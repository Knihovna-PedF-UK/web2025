\newcommand\testik[3][000000]{
  \pagecolor[HTML]{#2}%
  \AddToHookNext{ shipout/background } {
    \setgridorigin{0}{0}
    % logo pravém horním rohu
    \putongrid{15}{1}{
      \uklogo[0.12\textheight]
    }

    % obrázek v kroužku, s průmměrem 8 dílků gridu
    \putongrid{-2}{2}{\circleimage[\getgridunitx{8}]{celek}}

    % nastavení původu pro grid, aby se nám lépe počítalo umístění
    \setgridorigin{7}{2}
    % otevírací doba
    \putongrid{0}{0}{
      {\Huge\textcolor[HTML]{#3}{Otevírací doba}}
    }
      % \bigskip

    \bgroup
    \color[HTML]{#1}
    \putongrid{0}{2}{
      \strut Studovny v Rettigové\par\bigskip
      \begin{tabular}{@{}ll@{}}
        Po-Čt: & 8:00 - 18:00 \\
        Pá: & {8:00 - 17:00}\\
      \end{tabular}\par
    }

    \putongrid{5}{2}{{
      % \color{white}
      \strut Studovna v Celetné\par\bigskip
      \begin{tabular}{@{}ll@{}}
        Po-Čt: & 9:00 - 18:00 \\
        Pá: & 9:00 - 13:00\\
      \end{tabular}\par
    }}
    \egroup

    % zobrazit aktuální barvy
    \putongrid{-6}{5}{
      \textcolor[HTML]{000000}{{\tiny Nadpis: #3, pozadí: #2}}
    }

    % \putongrid{9}{2}{\parbox{\getgridunitx{3}}{\selectlanguage{latin}\tiny\lipsum[1]}}

  }

}
