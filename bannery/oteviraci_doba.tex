% calculate dimensions of the grid
\NewDocumentCommand\calculategriddims{mm}{
  \int_set:Nn \l_pedf_grid_cols { #1 }
  \int_set:Nn \l_pedf_grid_rows { #2 }
  \dim_set:Nn \l_pedf_grid_x { \paperwidth / #1 }
  \dim_set:Nn \l_pedf_grid_y { \paperheight / #2 }
}

\NewDocumentCommand\setgridorigin{mm}{
  \dim_set:Nn \l_pedf_origin_x { \l_pedf_grid_x * #1 }
  \dim_set:Nn \l_pedf_origin_y { -\l_pedf_grid_y * #2 }
}

% get dimensions for the grid unit

% #1 number of columns
\NewDocumentCommand\getgridunitx{m}{
  \dim_eval:n {\l_pedf_grid_x * #1}
}

% #1 number of rows
\NewDocumentCommand\getgridunity{m}{
  \dim_eval:n {\l_pedf_grid_y * #1}
}

% default grid dimensions
\calculategriddims{20}{10}
\setgridorigin{0}{0}

\NewDocumentCommand\circleobrazek{O{8cm}m}{%
  % našteme obrázek do boxu, aby se dal používat v l3draw příkazech a abysme mohli počítat jeho rozměry
  \hbox_set:Nn \l_tmpa_box {
    \includegraphics[height=#1]{#2}
  }
  \draw_begin:
    % nastavení ořezové cesty – kruh
    % průměr je dán výškou obrázku, takže se vejde celý
    \draw_path_circle:nn { 0 , 0 } { -0.5\box_ht:N \l_tmpa_box}
    % oříznutí podle cesty
    \draw_path_use_clear:n { clip }
    % musíme posunout box s obrázkem tak, aby jeho střed byl v místě středu kruhu, tedy v počátku souřadnic
    \draw_transform_shift:n {  -0.5\box_wd:N \l_tmpa_box , -0.5\box_ht:N \l_tmpa_box}
    % vykreslíme obrázek
    \draw_box_use:N \l_tmpa_box
  \draw_end:
}
